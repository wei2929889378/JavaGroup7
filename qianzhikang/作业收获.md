        // 点击导航，实现切换
        // key 表示的当前显示的是第⼏个
        (function (key) {
            // 获取所有的导航
            var navs = document.querySelectorAll('nav a');
            // 遍历 给导航 绑定事件，并且添加当前样式
            for (var i = 0; i < navs.length; i++) {
                // 如果是⽤户指定的当前样式
                if (key == i) {
                    navs[i].classList.add('active');
                    // 拿到要显示内容section的id
                    var secId = navs[i].dataset['cont'];
                    // 获取对应的section标签
                    document.querySelector('#' + secId).style.display = 'block';
                }
                // 给每⼀个导航绑定点击事件
                navs[i].onclick = function () {
                    // 之前有active样式的清除, 之前显示的section 隐藏
                    var currentNav = document.querySelector('.active');
                    // 获取对应的内容区域 ，让其隐藏
                    var currentId = currentNav.dataset['cont'];
                    // 去掉导航的active 样式
                    currentNav.classList.remove('active');
                    // 对应的内容区域
                    document.querySelector('#' + currentId).style.display = 'none';
                    // 突出显示⾃⼰ 导航添加样式 对应的section 显示
                    // 给⾃⼰添加active样式
                    this.classList.add('active');
                    // 对应的section模块显示出来
                    var myId = this.dataset['cont'];
                    document.querySelector('#' + myId).style.display = 'block';
                }
            }
        })(0);
    </script>
</body>

</html>

## 新知识：5：div的draggable属性

draggable属性设置为ture 可以使div可以拖动，并且拖动的同时，可以配合js的event进行控制

做出一些处理操作。

示例：

```html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>拖拽练习完整代码</title>
    <style>
        .one {
            width: 400px;
            height: 400px;
            border: 1px solid#aaa;
            margin-left: 200px;
            margin-top: 200px;
        }

        .one>div,
        .two>div {
            width: 98px;
            height: 98px;
            border: 1px solid#aaa;
            border-radius: 50%;
            background-color: rgb(103, 150, 179);
            float: left;
            text-align: center;
            line-height: 98px;
        }

        .two {
            width: 400px;
            height: 400px;
            border: 1px solid#aaa;
            position: absolute;
            left: 700px;
            top: 200px;
        }
    </style>
</head>

<body>

    <div class="one">
        <div draggable="true">1</div>
        <div draggable="true">2</div>
        <div draggable="true">3</div>
        <div draggable="true">4</div>
        <div draggable="true">5</div>
        <div draggable="true">6</div>
        <div draggable="true">7</div>
        <div draggable="true">8</div>
    </div>
    <div class="two"></div>
    <script>
        let boxes = document.querySelectorAll('.one div');// 临时的盒子: 用于存放当前拖拽的元素
        let two = document.querySelector('.two');
        let temp = null;// 8个小盒子分别绑定拖拽事件
        for (let i = 0, len = boxes.length; i < len; i++) {
            boxes[i].ondragstart = function () {
                // 保持当前拖拽的元素
                temp = this; console.log(temp);
            }
            boxes[i].ondragend = function () {
                // 当拖拽结束，清空temp
                temp = null; console.log(temp);
            }
        }// 目标元素的拖拽事件
        two.ondragover = function (e) {
            // 阻止拖拽的默认行为
            e.preventDefault();
        }//当在目标元素上松开鼠标是触发
        two.ondrop = function () {
            // 将拖拽的元素追加到 two里面来
            this.appendChild(temp);
        }
    </script>

</body>

</html>
```

## 实际开发常用：6.关于浏览器本地的存储

F12呼出Chrome的控制台，可以发现浏览器有本地存储的功能。基本都是一些键值对。通常用于记录用户在本页面的一些操作，如记住密码等等。

### LocalStorage 本地存储

本地存储将当前页面信息存储在浏览器中

![image-20211105102102125](https://pic-go.oss-cn-shanghai.aliyuncs.com/typora-img/202111051021164.png)



可以使用js将数据按照键值对的方式进行存储：

```js
 btns[0].onclick = function () {
            window.localStorage.setItem('userName', txt.value);
        }
```

这里对页面上的按钮进行绑定事件，当点击页面的时候将userName输入框中的数据进行本地存储

测试：

![image-20211105102903011](https://pic-go.oss-cn-shanghai.aliyuncs.com/typora-img/202111051029098.png)



### sessionStorage会话存储

只用于浏览器和服务端的一次会话，当关闭当前页面或关闭浏览器的时候，数据就会销毁，

相当于一个暂存区。

同样可以使用js进行控制存储：

```js
  btns[0].onclick = function () {
            window.sessionStorage.setItem('userName', txt.value);
            window.sessionStorage.setItem('pwd', '123456');
            window.sessionStorage.setItem('age', 18);
        }
```

效果：

![image-20211105103014269](https://pic-go.oss-cn-shanghai.aliyuncs.com/typora-img/202111051030358.png)

同样的我们也可以获取、更新、删除、清空当中的数据：

```js
        // sessionStorage获取数据
        btns[1].onclick = function () {
            txt.value = window.sessionStorage.getItem('userName');
        }
        // sessionStorage更新数据
        btns[2].onclick = function () {
            window.sessionStorage.setItem('userName', txt.value);
        }
        // sessionStorage删除数据
        btns[3].onclick = function () {
            window.sessionStorage.removeItem('userName');
        }
        // sessionStorage清空数据
        btns[4].onclick = function () {
            window.sessionStorage.clear();
        }
```

### 例：记住秘密

```html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>记住用户名和密码练习</title>
</head>

<body>
    <label for="">用户名：
        <input type="text" class="userName" /></label><br /><br />
    <label for="">密码：
        <input type="text" class="pwd" /></label><br /><br />
    <label for="">
        <input type="checkbox" class="check" id="" />记住密码</label><br /><br /><button>登录</button>
    <script>
        var userName = document.querySelector('.userName');
        var pwd = document.querySelector('.pwd'); 
    var chk = document.querySelector('.check'); 
    var btn = document.querySelector('button');
    // 当点击登录的时候如果勾选“记住密码”，就存储密码；否则就清除密码
    btn.onclick=function () {
        if (chk.checked) {
            // 记住数据
            window.localStorage.setItem('userName', userName.value);
            window.localStorage.setItem('pwd', pwd.value);      
        } else {
            // 清除数据
            window.localStorage.removeItem('userName');window.localStorage.removeItem('pwd');      
        }    
    }// 下次登录时，如果记录的有数据，就直接填充
    window.onload=function () {
        userName.value=window.localStorage.getItem('userName');
        pwd.value=window.localStorage.getItem('pwd');    
    }
    </script>
</body >
</html >
```
